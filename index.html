import pandas as pd
import numpy as np

# Loading messy raw business data
df = pd.read_csv("dirty_business_logs.csv")

# Executing cleaning routines
df.dropna(subset=["Revenue", "CustomerID"], inplace=True)
df["Region"] = df["Region"].str.strip().str.title()
df["Revenue"] = df["Revenue"].replace(r'[\$,]', '', regex=True).astype(float)

# Extracting actionable insights
kpis = {
    "Total_Sales": df["Revenue"].sum(),
    "Avg_Order_Val": df["Revenue"].mean(),
    "Active_Users": df["CustomerID"].nunique()
}

print("✨ Dashboard-ready dataset generated!")
print(f"Revenue tracked: ${kpis['Total_Sales']:,.2f}")
